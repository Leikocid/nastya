function fi(x:integer, y:integer):integer
{
 var z:integer;
 z = x * (x + y);
 return z; 
};  

function fs(a:string, b:integer):integer
{
 var c:integer;
 use function strlen(p:string):integer;
 c = strlen(a) + b;
 return c; 
};
